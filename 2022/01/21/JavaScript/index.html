问题：因为arguments获取所有的参数，如果我们想获取除传递以外的参数，如何办？

    for (let i = 2; i < arguments; i++) {
        //....   
    }
}
//现在rest==>   只能写在最后面，用...标识
function test(a,b,...rest) {
    console.log("a" + a);    
    console.log( "b" + b);    
    console.log(rest);
}

3.2 方法

方法即是把函数放在对象的里面，对象只有：属性和方法

var heng = {
    name : "yiheng",
    brith : '2000', 
    //方法    
    age : function () {
        var now = new Date().getFullYear(); 
        return now - this.brith;  //this默认指向调用它的对象    
    }
}
//调属性
heng.name;
//调方法
heng.age();

在js中，可以控制this的指向 ===> 通过apply,所有的函数均有

//可将方法定义在外面
function getAge() {
    var now = new Date().getFullYear();   
    return now - this.brith;
}
var heng = { 
    name : "yiheng",  
    brith : '2000',    
    //方法    
    age : getAge}
//调用调用    
heng.age() //打印22    
getAge()  //打印NaN  this并没有指向heng    
getAge.apply(heng,[]);  //此时this指向了heng，打印22

4.2 JSON

JSON（JavaScript Object Notation）是一种轻量级资料交换格式,采用完全独立于编程语言的文本格式来存储和表示数据。简洁和清晰的层次结构使得 JSON 成为理想的数据交换语言。 易于人阅读和编写，同时也易于机器解析和生成，并有效地提升网络传输效率

在js中，一切均对象，任何js支持的类型均可以用JSON进行表示

格式：

  • 对象都用 {}
  • 数组用 []
  • 所有的键值对 用key : value
var obj = {
    name:"heng", 
    age : 20,    sex : '男'
}
//对象转json字符串
JSON.stringify(obj)
'{"name":"heng","age":20,"sex":"男"}'
// json字符串解析为对象
JSON.parse('{"name":"heng","age":20,"sex":"男"}')
{name: 'heng', age: 20, sex: '男'}
