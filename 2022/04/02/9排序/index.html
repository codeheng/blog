//对数组长度为n的arr排序
void insert_sort(int *arr,int n) {
     //进行n-1轮
    for (int i = 1;i < n;i++) {
        for (int j = i; j > 0 && arr[j] < arr[j-1];j--) {
            swap(arr[j],arr[j-1]);
        }
    }
    return;
}

冒泡

void bubble_sort(int *arr,int n) {
    int flag = 1;//用于优化冒泡
    //进行n-1轮循环
    for (int i = 1;i < n && flag != 0;i++) {
        flag = 0;
        for (int j = 0;j < n-i;j++) {
            if (arr[j] <= arr[j+1]) continue;
            swap(arr[j],arr[j+1]);
            flag ++;
        }
    }
    return;
}

归并

void merge_sort(int *arr,int l,int r) {
    //若元素太少，< 2
    if ( r - l < 1) {
        //此时两个元素,并且前面值大，则进行交换
        if (r -l == 1 && arr[l] > arr[r]) swap(arr[l],arr[r]);
        return;
    }
    //保存中间结点
    int mid = (l+r) >> 1;
    //两边分别递归，进行排序
    merge_sort(arr,l,mid);
    merge_sort(arr,mid+1,r);
    //递归排序完，两两有序，进行归并过程
    //开辟新空间
    int *temp = (int*)malloc(sizeof(int) * (r-l+1));
    //p1指向第一个数组第一个位置,p2指向归并的第二个数组的第一个位置，k指向新开辟的的第一个位置
    int p1 = l,p2 = mid + 1,k = 0;
    //当一个数组或者第二个数组有元素时,进行循环
    while (p1 <= mid || p2 <= r) {
        //归并，放到临时存储区
        //当 第二个数组为空，或者第一个不空且满足第一个值小 p1指向进行赋值，否则p2指向赋值
        if (p2 > r || p1 <= mid && arr[p1] <= arr[p2])  {
            temp[k++] = arr[p1++];
        }
        else {
            temp[k++] = arr[p2++];
        }
    }
    //此时已有序，再将数据拷贝到原数组中
    //复制temp指向内存内容的前 sizeof(int)*(r-l+1) 个字节到arr+l所指的内存地址上
    memcpy(arr+l,temp,sizeof(int) * (r-l+1));

}

选择排序

void select_sort(int *arr,int n) {
    for (int i = 0;i < n-1;i++) {
        int index = i;
        for (int j = i+1;j < n;j++) {
            if (arr[index] > arr[j]) index = j;
        }
        swap(arr[i],arr[index]);
    }
    return;
}

快速排序

void quick_sort(int *arr,int l,int r) {
    if (r <= l) return;
    //x指向头，y指向尾，index为基准值，选择第一个为基准
    int x = l,y = r,index = arr[l]; 
    //若左边位置 < 右边位置 一直循环
    while (x < y) {
        //循环进行寻找，至 右边对应值>基准值,则y减小
        while (x < y && arr[y] >= index) --y;
        //此时若满足 左 < 右,则将右边值放到左边，则x对应位置自增
        if (x < y) arr[x+=] = arr[y];
        //同理操作左边
        while (x < y && arr[x] <= index) ++x;
        if (x < y) arr[y--] = arr[x];
    }
    //此时头尾相交,赋值，然后左右分别进行快排
    arr[x] = index;
    quick_sort(arr,l,x-1);
    quick_sort(arr,x+1,r);
    return;
}
