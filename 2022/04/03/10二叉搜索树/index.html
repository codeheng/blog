二叉搜索树

简述

二叉查找树（BST：Binary Search Tree）是一种特殊的二叉树，它改善了二叉树节点查找的效率

  • 若任意节点的左子树不空，则左子树上所有节点的值均小于它的根节点的值
  • 若任意节点的右子树不空，则右子树上所有节点的值均大于它的根节点的值；
  • 没有键值相等的节点

左 < 根 < 右

代码

定义

typedef struct Node {
    int data;
    struct Node *lchild,*rchild;
}Node;

获得新结点和销毁

//获得结点
Node *getNewNode(int data) {
    Node *p = (Node *)malloc(sizeof(Node));
    p->data = data;
    p->rchild = p->lchild = NULL;
    return p;
}
//销毁
void clear(Node *node) {
    if (node == NULL) return;
    clear(node->lchild);
    clear(node->rchild);
    free(node);
    return;
}

插入

//向root为根的结点中，插入值key
Node *insert(Node *root,int key) {
    //若root为空，直接返回
    if (root == NULL) return getNewNode(key);
    //对应值为key相等，返回root
    if(root->data == key) return root;
    //对应值大于key，向左进行插入
    if (root->data > key) root->lchild = insert(root->lchild,key);
    //否则向右进行插入
    else root->rchild = insert(root->rchild,key);
    //最终返回root
    return root;
}

删除

//找前驱
Node *prodecessor(Node *root) {
    Node *temp = root->lchild;
    //找左子树最右结点
    while (temp->rchild) temp = temp->rchild;
    return temp;
}
//删除
Node *delete(Node *root,int key) {
    //先进行寻找
    if (root == NULL) return root;
    //向左进行查找
    if (root->data > key) root->lchild = delete(root->lchild,key);
    //否则向右找
    else if(root->data < key) root->rchild = delete(root->rchild,key);
     //此时当前结点即要删除结点,讨论度0,1,2
    else {
        //度为0
        if (root->lchild == NULL && root->rchild == NULL) {
            free(root);
            return NULL;
        }
        //度为1,下面这段逻辑，也满足删除度为0需求，上方可以去掉
        else if (root->lchild == NULL || root->rchild == NULL) {
            //看哪边不空，然后进行暂存
            Node *temp = root->lchild ? root->lchild : root->rchild;
            free(root);
            return temp;
        }
        //度为2
        else {
            //找到度为2的前驱或后继，此处找前驱
            Node *temp = prodecessor(root);
            //找到后，进行值的替换
            root->data = temp->data;
            //在左子树中删除数据为temp->data的结点
            root->lchild = delete(root->lchild,temp->data);
        }
    }
    return root; //代表处理完之后的结点根地址
}

//中序遍历
void inOrder(Node *root) {
    if (root == NULL) return;
    inOrder(root->lchild);
    printf("%d ",root->data);
    inOrder(root->rchild);
    return;
}

    int op,val;
    Node *root = NULL;
    while ( scanf("%d%d",&op,&val) != EOF ) {
        switch (op) {
            //1代表插入，2代表删除
            case 1:{
                root = insert(root,val);
            }
                break;
            case 2: {
                root = delete(root,val);
            }
                break;
        }
        printf("inOrder output : ");

        printf("\n");
    }
    clear(root);
    return 0;
}
