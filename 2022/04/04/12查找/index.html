二分查找

//解决有序数组的问题 如: 1 2 3 5 8 9 
int binary_search1(int *arr,int n,int x) {
    int head = 0,tail = n - 1,mid;
    while (head <= tail) {
        //获得中间的值 = (head + tail) /2
        mid = (head + tail ) >> 1;
        //若中间值=目标值，直接返回
        if (arr[mid] == x) return mid;
        //若中间值 > 目标值，则表明要向左找，否则向右找
        if (arr[mid] > x) tail = mid - 1;
        else head = mid + 1;
    }
    //返回-1表示查找失败
    return -1;
}

//解决前面都是1后面为0的问题,找到最后1的位置，如11111000000
int binary_search2(int *arr,int n) {
    int head = 0,tail = n-1,mid;
    //只剩一个元素时，即找到了最后的1,故 无等号
    while (head < tail) {
        //指向两个元素中，较靠后的那个
        mid = (head + tail) >> 1;
       // printf("head:%d,tail:%d,mid:%d\n",head,tail,mid);
        if (arr[mid] == 1) head = mid;
        else tail = mid - 1;
    }
    return head;
}
//解决如：00000011111  找到第一个1
int binary_search3(int *arr,int n) {
    int head = 0,tail = n,mid;
    while (head < tail) {
        mid = (tail + head) >> 1;
        if (arr[mid] == 1) tail = mid;
        else head = mid + 1;
    }
    return head == n ? -1 : head;
}

哈希表

概述

散列表（Hash table，也叫哈希表），是根据关键码值(Key value)而直接进行访问的数据结构。

也就是说，它通过把关键码值映射到表中一个位置来访问记录，以加快查找的速度。

这个映射函数叫做散列函数，存放记录的数组叫做散列表。
给定表M，存在函数f(key)，对任意给定的关键字值key，代入函数后若能得到包含该关键字的记录在表中的地址，则称表M为哈希表，函数f(key)为哈希函数

  • 在列表查找中，使用最广泛的二分查找算法，复杂度为O(log2n)，但其始终只能用于有序列表。

  • 普通无序列表只能采用遍历查找，复杂度为O(n)

  • 较为理想的哈希函数实现的哈希表，对其任意元素的查找速度始终为常数级，即O(1)

哈希碰撞(冲突)

  • 散列法:开放寻址法和再散列法
  • 拉链法

应用

工程上，经常用于通过名称指定配置信息、通过关键字传递参数、建立对象与对象的映射关系等。

目前最流行的NoSql数据库之一Redis，整体的使用了哈希表思想

代码

定义

typedef struct Node {
    char *str;//存储字符
    struct Node *next;//指向下一个的指针
}Node;

typedef struct HashTable {
     //存储若干个链表头地址的数组
    //哈希表(也是顺序表)里面存储的是一个地址 地址所指向的类型是Node*类型
    Node **data;
    //哈希表大小
    int size;
}HashTable;

初始化

//初始化结点
Node *init_node(char *str,Node *head) {
    //分配结点空间
    Node *p = (Node *)malloc(sizeof(Node));
     //申请新的存储空间，将str的内容拷贝到其中,返回新的地址
    p->str = strdp(str);
    p->next = head;
    return p;
}
//初始化n个元素的哈希表
HashTable *init_hashtable(int n) {
    //为哈希表分配空间
    HashTable* h = (HashTable*)malloc(sizeof(HashTable));
    //让空间大一些
    h->size = n * 2;
    //初始化存储地址的数组,并每个字节初始化为0,故用calloc
    h->data = (Node **)calloc(sizeof(Node*),h->size);
    return h;
}
  • char * strdup(const char *s); 包含在#include <string.h>

strdup()会先用maolloc()配置与参数s 字符串相同的空间大小，然后将参数s 字符串的内容复制到该内存地址，然后把该地址返回。该地址最后可以利用free()来释放

  • void* calloc (size_t num, size_t size); 包含在#include <stdlib.h>

在内存中动态地分配 num 个长度为 size 的连续空间，并将每一个字节都初始化为 0,分配成功返回指向该内存的地址，失败则返回 NULL

calloc() 在动态分配完内存后，自动初始化该内存空间为零

而 malloc() 不初始化，里边数据是未知的垃圾数据

销毁

//销毁
void clear_node(Node *node ) {
    if (node == NULL) return;
    //先将node进行保存
    Node *p = node,*q;
    while (p) {
        //利用q指针记录下一个位置
        q = p->next;
        //释放原先的str位置
        free(p->str);
        //再释放p
        free(p);
        //当前节点指向下一个位置
        p = q;
    }
    return;
}
//销毁哈希表
void clear_hashtable(HashTable *h) {
    if (h == NULL) return;
    //清除每一个数据区，均是个Node类型指针
    for (int i  = 0; i < h->size;i++) {
        clear_node(h->data[i]);
    }
    //释放指向Node 数组的指针
    free(h->data);
    //再释放整个哈希表
    free(h);
    return;
}

//哈希函数
int BKDRHash(char *str) {
    int seed = 31,hash = 0;
    for (int i = 0;str[i];i++) hash = hash * seed + str[i];
    //0x7fffffff 表示的是符号位为0 其余位为1的数 
    return hash & 0x7fffffff;
}

//哈希表插入,将str字符串插入哈希表中

查找

int search(HashTable *h,char *str) {
    //获得哈希值
    int hash = BKDRHash(str);
    //获得索引
    int index = hash % h->size;
    //将内容结点暂存
    Node *p = h->data[index];
    //若不为空，且内部存的字符串和要传入的字符串不相等，则顺着链表继续找
    while (p && strcmp(p->str,str)) p = p->next;
    return p != NULL;
}

#include <stdio.h>
#include <string.h>
int main() {
    //表示何种操作
    int op;

    char str[100];
    //初始化哈希表,100个元素
    HashTable *h = init_hashtable(100);

    while ( scanf("%d%s",&op, str) != EOF ) {
        switch (op) {
            //0代表插入，1代表查找

                printf("insert %s to hashtable\n",str);
                insert(h,str);
            }
                break;
            case 1: {
                printf("search %s str result = %d\n",str,search(h,str));
            }
                break;

    }

}
