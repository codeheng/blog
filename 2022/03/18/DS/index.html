顺序表

定义

typedef struct Vector 
{
    int *data; //数据域
    int size,length; //线性表总大小和当前的长度
}Vector;

初始化

void *init(int n) 
{
    Vector *vec = (Vector*)malloc(sizeof(Vector)); //开辟结构新的空间
    vec->data = (int *)malloc(sizeof(int) * n); //数据域开辟空间
    vec->size = n;
    vec->length = 0;
    return vec;
}

增加元素

int insert(Vector *vec,int index,int value)
{
    //条件判断，若线性表为空或位置不符，则直接return
    if (vec == NULL) return 0;
    if (index < 0 || index > vec->length) return 0;
    //若线性表长度达到总长度，则进行扩容操作
    if (vec->length == vec->size)
    {
        //扩容失败，直接返回
        if (! expand(vec)) return 0;
        printf("expand vector size to %d suceess\n",vec->size);
    }
    //插入逻辑
    for (int i = vec->length;i > index;i--)
    {
        vec->data[i] = vec->data[i-1];
    }
    vec->data[index] = value;//将值进行插入
    vec->length ++; //线性表长度要+1 (别忘记
    return 1;
}
//扩容操作
int expand(Vector *vec) 
{
    int new_size = vec->size * 2;//进行二倍扩容
    int *p = (int *)realloc(vec->data,sizeof(int) * new_size);//重新申请空间
    if (p == NULL) return 0; //扩容失败
    vec->size = new_size;
    vec->data = p;
    return 1;
}

realloc

#include <stdlib.h>
void *realloc( void *ptr, size_t size );

函数将ptr 对象的储存空间改变为给定的大小size。 参数size可以是任意大小，大于或小于原尺寸都可以。 返回值是指向新空间的指针，如果错误发生返回NULL

试着在原先的基础先进行扩容，若成功，返回之前那个区域的首地址

若失败，则会调用malloc，另外申请一片空间，并把之前数据拷贝，返回新的首地址，并释放之前的空间

所以不可以像下面这样写

vec->size *= 2;
vec->data = (int *)realloc(vec->data,sizeof(int) * size);

若返回空地址，则之前的内容丢失了，因为被赋值为NULL

即造成内存泄漏 ==> 你开辟的这个空间，但是你用不了了，把内存信息丢失了

删除元素

int delete(Vector *vec,int index)
{
    //正确性检查
    if (vec == NULL || vec->length == 0) return 0;
    if (index < 0 || index >= vec->length) return 0;
    //删除
    for (int i = index + 1;i < vec->length; i++)
    {
        vec->data[i - 1] = vec->data[i];
    }
    vec->length --;
    return 1;
}

销毁线性表

void clear(Vector *vec) 
{
    if (vec == NULL) return;
    free(vec->data); //释放数据域
    free(vec); //释放整个结构体
}

输出线性表

void output(Vector *vec)
{
    printf("Vector(%d) = [", vec->length);
    for (int i = 0; i < vec->length;i++ ) {
        if (i != 0) printf(", "); // 进行分隔
        printf("%d",vec->data[i]);
    }
    printf("]\n");
    return;
}

测试

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#define MAX_OP 10 //执行十次操作 (增or删)
int main() {
    srand(time(0)); //随机数种子，让其随机产生数字
    Vector *vec = init(1); //让线性表初始最大为1
    int op,ind,val;
    for (int i = 0; i < MAX_OP;i++) {
        op = rand() % 4; //op表示增还是删的概率，让增加概率为3/4,删1/4
        ind = rand() % (vec->length + 3); //索引的位置,随机产生
        val = rand() % 100; //产生1-100的数字
        switch (op) {
            case 2:
            case 3:
            case 0: {
                printf("insert %d at %d to vector = %d\n",
                       val, ind, insert(vec, ind, val));
            }
            break;
            case 1: {
                printf("delete item at %d from vector = %d\n",
                       ind, delete(vec, ind));
            }
            break;
        }
        output(vec);
        printf("\n");

    }
    return  0;
}
