二叉树

简述

术语

  • 树的结点（node）：包含一个数据元素及若干指向子树的分支；
  • 孩子结点（child node）：结点的子树的根称为该结点的孩子；
  • 双亲结点：B 结点是A 结点的孩子，则A结点是B 结点的双亲；
  • 兄弟结点：同一双亲的孩子结点；堂兄结点：同一层上结点；
  • 祖先结点: 从根到该结点的所经分支上的所有结点
  • 子孙结点：以某结点为根的子树中任一结点都称为该结点的子孙
  • 结点层：根结点的层定义为1；根的孩子为第二层结点，依此类推；
  • 树的深度：树中最大的结点层
  • 结点的度：结点子树的个数
  • 树的度：树中最大的结点度。
  • 叶子结点：也叫终端结点，是度为 0 的结点；
  • 分枝结点：度不为0的结点；

性质

  • 在二叉树中，第 i 层上至多有2^(i-1)个节点（i ≥ 1）
  • 深度为k的二叉树至多有2^(k−1)个节点（k≥1）
  • 对一棵二叉树，如果叶子节点的个数为n0，度为2的节点个数为n2，则n0=n2+1
  • 具有n个节点的完全二叉树的深度为⌊log2n⌋+1

代码实现

定义

//节点的定义
typedef struct Node {
    int val;
    struct Node *lchild,*rchild;
}Node;
//树定义
typedef struct Tree {
    Node *root;
    int n; //代表多少个结点
}Tree;

初始化

//结点初始化
Node *getNewNode(int val) {
    Node *p = (Node *) malloc(sizeof(Node));
    p->val = val;
    p->lchild = p->rchild = NULL;
    return p;
}
//树初始化
Tree *getNewTree() {
    Tree *tree = (Tree*)malloc(sizeof(Tree));
    tree->n = 0;
    tree->root = NULL;
    return tree;
}

输出

//输出节点
void outputNode(Node *node) {
    if (node == NULL) return;
    printf("%d",node->val);
    if (node->lchild == NULL && node->rchild == NULL) return;
    printf(" ( ");
    outputNode(node->lchild);
    printf(",");
    outputNode(node->rchild);
    printf(" )");
    return;
}
//将树转为广义表输出
void outputTree(Tree *tree) {
    printf("Tree(%d) = ",tree->n);
    outputNode(tree->root);
    printf("\n");
    return;
}

插入节点

插入后，树的节点肯定增加，故要利用传出参数

//插入节点,ret为传出参数
Node *insertNode(Node *root,int val,int *ret) {
    //若为空，创建新节点
    if (root == NULL) {
        *ret = 1; //为1代表插入成功
        return getNewNode(val); //创建节点
    }
    if (root->val == val) return root;
    if (root->val > val) root->lchild = insertNode(root->lchild,val,ret);
    else root->rchild = insertNode(root->rchild,val,ret);
    return root;
}
//在树中进行插入,即二叉排序树
void insert(Tree *tree,int val) {
    int flag = 0;//标志位，开始为0，若成功，则将传出参数为1带回，然后自增！
    tree->root = insertNode(tree->root,val,&flag);
    tree->n += flag;//将树中结点数目进行更新
    return;
}

遍历

//前序遍历
void preOrderNode(Node *pNode) {
    if (pNode == NULL) return;
    printf(" %d",pNode->val);
    preOrderNode(pNode->lchild);
    preOrderNode(pNode->rchild);
}
void preOrder(Tree *pTree) {
    printf("preOrder: " );
    preOrderNode(pTree->root);
    printf("\n");
    return;
}
//中序遍历
void inOrderNode(Node *pNode) {
    if (pNode == NULL) return;
    inOrderNode(pNode->lchild);
    printf(" %d",pNode->val);
    inOrderNode(pNode->rchild);
}
void inOrder(Tree *pTree) {
    printf("inOrder: " );
    inOrderNode(pTree->root);
    printf("\n");
    return;
}
//后序遍历
void postOrderNode(Node *pNode) {
    if (pNode ==NULL) return;
    postOrderNode(pNode->lchild);
    postOrderNode(pNode->rchild);
    printf(" %d",pNode->val);
}
void postOrder(Tree *pTree) {
    printf("postOrder: " );
    postOrderNode(pTree->root);
    printf("\n");
    return;
}

main测试

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
int main() {
    srand(time(0));
    Tree *tree = getNewTree(); //初始化树
    //共五个节点，利用随机数生成
    for (int i = 0; i < 5;i++) {
        int val = rand() % 100;
        insert(tree,val); //插入节点
        outputTree(tree);
    }
    preOrder(tree); //前序
    inOrder(tree); //中序
    postOrder(tree); //后序
    clearTree(tree); //销毁
    return 0;
}
